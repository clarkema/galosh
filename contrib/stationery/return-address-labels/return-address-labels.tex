%%
%% Building this document requires that you have a working LaTeX system
%% installed; ideally one that provides pdflatex(1).
%%
%% Given that, the easiest way to build a document is:
%%
%% $ pdflatex return-address-labels.tex
%%
%% The settings below are suitable for printing on Avery J8659 Quick Dry
%% Mini Inkjet Labels (270 to a page of A4.)  To adapt for other label
%% layouts, just modify the count, margin, and border variables below.
%%

%% To print on US-style Letter paper rather than A4, comment out the line
%% containing 'a4paper' below, and uncomment the line containing 'letter'.
\documentclass[a4paper]{article}
%% \documentclass[letter]{article}

\usepackage[newdimens]{labels}

\LabelCols=10           % 10 columns of labels
\LabelRows=27           % 27 rows of labels

\LeftPageMargin=5mm
\RightPageMargin=4mm
\TopPageMargin=13mm
\BottomPageMargin=13mm

\InterLabelColumn=2mm
\InterLabelRow=0mm

\LeftLabelBorder=1mm
\RightLabelBorder=1mm
\TopLabelBorder=1mm
\BottomLabelBorder=1mm
% \LabelGridtrue        % Uncomment this line to print a box *outside* the
                        % label area for the purposes of testing alignment.

\numberoflabels=270     % Total number of labels to print.  You can print
                        % less than a full page by adjusting this.

\begin{document}
\addresslabel[\tiny]
% Modify the line below to reflect your address.  Using a ~ instead of
% a literal space will tend to lead to better layout in this case.
{\textbf{Name}\\Addr~1\\Addr~2\\Country.}
\end{document}
